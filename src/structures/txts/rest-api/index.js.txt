'use strict';

import 'dotenv/config';
import './src/structures/I18n.js';

import Database from './src/structures/Database.js';
import Models from './src/models/Model.js';
import Server from './src/Server.js';

(async() => {
  await Database.authenticate();
    await Models.loadModels(Database);
    await Database.sync({ force: true });
    await Models.loadModelsAssociate(Database);
  return new Server(Database).start();
})();
